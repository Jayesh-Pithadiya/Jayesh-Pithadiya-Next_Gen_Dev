import java.util.Scanner;
import java.util.Timer;
import java.util.TimerTask;

class QuizGame {
    private QuizQuestion[] questions;
    private int score;

    public QuizGame(QuizQuestion[] questions) {
        this.questions = questions;
        this.score = 0;
    }

    public void start() {
        Scanner scanner = new Scanner(System.in);

        for (int i = 0; i < questions.length; i++) {
            System.out.println("\nQuestion " + (i + 1) + ":");
            System.out.println(questions[i].getQuestion());

            String[] options = questions[i].getOptions();
            for (int j = 0; j < options.length; j++) {
                System.out.println((j + 1) + ": " + options[j]);
            }

            // Start the timer for 10 seconds
            Timer timer = new Timer();
            TimerTask task = new TimerTask() {
                @Override
                public void run() {
                    System.out.println("Time is up!");
                    scanner.close();
                    System.exit(0);
                }
            };
            timer.schedule(task, 10000); // 10 seconds

            System.out.print("Choose an option (1-" + options.length + "): ");
            int answerIndex = scanner.nextInt() - 1;

            timer.cancel(); // Stop the timer if answered in time

            if (questions[i].isCorrectAnswer(answerIndex)) {
                System.out.println("Correct!");
                score++;
            } else {
                System.out.println("Incorrect! The correct answer was: " + (questions[i].isCorrectAnswer(questions[i].correctAnswerIndex) ? options[questions[i].correctAnswerIndex] : ""));
            }
        }

        displayResults();
        scanner.close();
    }

    private void displayResults() {
        System.out.println("\nQuiz Over!");
        System.out.println("Your score: " + score + "/" + questions.length);
    }
}
